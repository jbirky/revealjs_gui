# integration example
# https://docs.scipy.org/doc/scipy/reference/generated/scipy.integrate.odeint.html
   
import numpy as np
from scipy.integrate import odeint

def pend(y, t, b, c):
    theta, omega = y
    dydt = [omega, -b*omega - c*np.sin(theta)]
    return dydt

b = 1.0
c = 3.0
y0 = [np.pi - 0.1, 0.0]
t = np.linspace(0, 10, 101)
   
sol = odeint(pend, y0, t, args=(b, c))
